using tales

class Build : TalesBuildPod
{
		
  override Void setProperties(){
	File f := File( (typeof->sourceFile)->toUri)
	podName = f.parent.name
	summary    = "The Cool $podName app"
	depends    = ["sys 1.0+", "util 1.0+","web 1.0+", "concurrent 1.0+", "tales 1.0+", "email 1.0+", "sql 1.0+"]
  }	

}
